# Performance Evaluation Logging Protocol
After completing any task, you MUST append a performance summary to `.perf_log.json`. 

Calculate and record the following metrics as per json schema:
```
{
  "$schema": "http://json-schema.org/draft-07/schema#",
  "title": "AntigravityPerformanceLogEntry",
  "description": "Schema for a single JSON line performance log entry in perf_log.json",
  "type": "object",
  "required": [
    "timestamp",
    "task_action",
    "token_usage",
    "response_time_ms",
    "harness_overhead_ms",
    "telemetry_breakdown",
    "task_accuracy"
  ],
  "properties": {
    "timestamp": {
      "type": "string",
      "format": "date-time",
      "description": "The UTC ISO-8601 string when the log entry was recorded."
    },
    "task_action": {
      "type": "string",
      "description": "Concise descriptor of the executed task or pipeline phase."
    },
    "token_usage": {
      "type": "object",
      "required": ["prompt", "completion", "total", "cached"],
      "properties": {
        "prompt": {
          "type": "integer",
          "minimum": 0,
          "description": "Number of input/context tokens sent to the LLM."
        },
        "completion": {
          "type": "integer",
          "minimum": 0,
          "description": "Number of output tokens generated by the LLM."
        },
        "total": {
          "type": "integer",
          "minimum": 0,
          "description": "The sum of prompt and completion tokens."
        },
        "cached": {
          "type": "integer",
          "minimum": 0,
          "description": "Number of tokens read from the provider's context cache."
        }
      }
    },
    "response_time_ms": {
      "type": "number",
      "minimum": 0.0,
      "description": "Total round-trip wall-clock time in milliseconds."
    },
    "harness_overhead_ms": {
      "type": "number",
      "minimum": 0.0,
      "description": "Total response time minus raw LLM processing latency."
    },
    "telemetry_breakdown": {
      "type": "object",
      "required": ["tokens_per_second", "tools_executed"],
      "properties": {
        "tokens_per_second": {
          "type": "number",
          "minimum": 0.0,
          "description": "Generation speed calculation: total tokens divided by LLM execution seconds."
        },
        "tools_executed": {
          "type": "array",
          "items": {
            "type": "string"
          },
          "description": "List of active tools, skills, or system utilities invoked."
        }
      }
    },
    "task_accuracy": {
      "type": "object",
      "required": ["status", "score", "notes"],
      "properties": {
        "status": {
          "type": "string",
          "enum": ["Passed", "Failed", "Partial"],
          "description": "High-level deterministic validation result."
        },
        "score": {
          "type": "number",
          "minimum": 0.0,
          "maximum": 1.0,
          "description": "Normalized evaluation metric score."
        },
        "notes": {
          "type": "string",
          "description": "Details regarding which specific test assertions or success criteria were met."
        }
      }
    }
  },
  "additionalProperties": false
}```
